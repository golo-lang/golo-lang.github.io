010package gololang;
011
012import java.net.URL;
013import java.io.InputStream;
014
015import org.eclipse.golo.compiler.GoloClassLoader;
016import org.eclipse.golo.cli.command.spi.CliCommand;
017
018import static java.util.Objects.requireNonNull;
019import static gololang.IO.toURL;
020
021/**
022 * A utility module containing functions to interact with the runtime environment.
023 */
024public final class Runtime {
025  private Runtime() {
026    // utility class
027  }
028
029  /**
030   * Get a boolean value from a system property or an environment variable.
031   *
032   * <p>Equivalent to {@code loadBoolean("true", prop, env, defaultValue)}
033   *
034   * @param prop system property to get the value from, ignored if {@code null}.
035   * @param env environment variable to get the value from, ignored if {@code null}.
036   * @param defaultValue value to return if neither the property nor the environment are defined.
037   */
038  public static boolean loadBoolean(String prop, String env, boolean defaultValue) {
039    return loadBoolean("true", prop, env, defaultValue);
040  }
041
042  /**
043   * Get a boolean value from a system property or an environment variable.
044   *
045   * <p>Checks if the property is defined and equals to the value (case insensitive), otherwise checks the environment
046   * variable, then use the default value.
047   *
048   * @param value content of the property.
049   * @param prop system property to get the value from, ignored if {@code null}.
050   * @param env environment variable to get the value from, ignored if {@code null}.
051   * @param defaultValue value to return if neither the property nor the environment are defined.
052   * @throws NullPointerException if {@code value} is null and one of the property or environment is defined.
053   */
054  public static boolean loadBoolean(String value, String prop, String env, boolean defaultValue) {
055    String val = null;
056    if (prop != null) {
057      val = System.getProperty(prop);
058      if (val != null) {
059        return val.toLowerCase().equals(value.toLowerCase());
060      }
061    }
062    if (env != null) {
063      val = System.getenv(env);
064      if (val != null) {
065        return val.toLowerCase().equals(value.toLowerCase());
066      }
067    }
068    return defaultValue;
069  }
070
071  private static boolean debug = loadBoolean("golo.debug", "GOLO_DEBUG", false);
072  private static boolean showStackTrace = loadBoolean("golo.debug.trace", "GOLO_DEBUG_TRACE", true);
073
074  /**
075   * Checks if debug mode is active.
076   *
077   * <p>The initial value is false, but can be redefined using the {@code golo.debug} system property, or the
078   * {@code GOLO_DEBUG} environment variable.
079   */
080  public static boolean debugMode() {
081    return debug;
082  }
083
084  /**
085   * Defines debug mode.
086   */
087  public static void debugMode(boolean v) {
088    debug = v;
089  }
090
091  /**
092   * Checks if the main golo command should print stack traces on error.
093   */
094  public static boolean showStackTrace() {
095    return showStackTrace;
096  }
097
098  /**
099   * Defines if the main golo command should print stack traces on error.
100   */
101  public static void showStackTrace(boolean v) {
102    showStackTrace = v;
103  }
104
105  private static CliCommand command;
106
107  /**
108   * Defines the golo command currently used.
109   *
110   * <p>This command is called at init time and should not be called in userland code.
111   */
112  public static void command(CliCommand cmd) {
113    command = requireNonNull(cmd);
114  }
115
116  /**
117   * Returns the golo command currently used.
118   */
119  public static CliCommand command() {
120    return command;
121  }
122
123  /**
124   * Returns the current thread class loader.
125   * <p>
126   * Possibly wrapped in a {@code GoloClassLoader} if necessary.
127   */
128  public static GoloClassLoader classLoader() {
129    ClassLoader cl = Thread.currentThread().getContextClassLoader();
130    if (cl instanceof GoloClassLoader) {
131      return (GoloClassLoader) cl;
132    }
133    return new GoloClassLoader(cl);
134  }
135
136  /**
137   * Load a golo file given its path or URL.
138   *
139   * <p>Equivalent to {@code load(file, true)}.
140   *
141   * @param file can be a {@code String}, a {@code java.nio.file.Path}, a {@code java.net.URL} or a {@code java.net.URI}
142   *   identifying the file to load.
143   * @return the loaded module class.
144   * @see #load(Object, boolean)
145   */
146  public static Class<?> load(Object file) throws Exception {
147    return load(file, true);
148  }
149
150  /**
151   * Load a golo file given its path or URL.
152   *
153   * <p>If {@code failOnException} is false, any exception is swallowed and null is return.
154   *
155   * @param file can be a {@code String}, a {@code java.nio.file.Path}, a {@code java.net.URL} or a {@code java.net.URI}
156   *   identifying the file to load.
157   * @param failOnException re-raise exception or not
158   * @return the loaded module class, or {@code null} if an error occurred and {@code failOnException} is {@code false}
159   * @see #load(Object, boolean)
160   */
161  public static Class<?> load(Object file, boolean failOnException) throws Exception {
162    try {
163      URL url = toURL(file);
164      try (InputStream is = url.openStream()) {
165        return classLoader().load(file.toString(), is);
166      }
167    } catch (Exception e) {
168      if (failOnException) {
169        throw e;
170      }
171      return null;
172    }
173  }
174
175  /**
176   * Returns the current golo version.
177   */
178  public static String version() {
179    return org.eclipse.golo.cli.command.Metadata.VERSION;
180  }
