013import java.lang.invoke.MethodHandle;
014import java.lang.invoke.MethodHandles;
015import java.lang.invoke.MethodType;
016import java.util.HashMap;
017import java.util.Map;
018import java.util.Set;
019import java.util.List;
020import java.util.LinkedList;
021import java.util.Objects;
022
023import static java.lang.System.arraycopy;
024import static java.lang.invoke.MethodType.genericMethodType;
025import static java.lang.invoke.MethodType.methodType;
026import static gololang.Predefined.isClosure;

028/**
029 * A dynamic object is an object whose properties can be dynamically added, changed and removed. Properties can be any
030 * object value or a function reference.
031 * <p>
032 * The methods <code>plug</code> and <code>propertyMissing</code> are left undocumented. They are being used
033 * by the Golo runtime to dispatch method invocations on dynamic objects.
034 */
035public final class DynamicObject {
036
037  private final Object kind;
038  private final HashMap<String, Object> properties = new HashMap<>();
039  private boolean frozen = false;
040
041
042  public DynamicObject() {
043    this("DynamicObject");
044  }
045
046  public DynamicObject(Object kind) {
047    this.kind = kind;
048  }
049
050  public boolean hasKind(Object k) {
051    return Objects.equals(kind, k);
052  }
053
054  public boolean sameKind(DynamicObject other) {
055    return Objects.equals(kind, other.kind);
056  }
057
058  @Override
059  public String toString() {
060    List<String> props = new LinkedList<>();
061    for (Map.Entry<String, Object> prop : properties.entrySet()) {
062      if (!isClosure(prop.getValue())) {
063        props.add(String.format("%s=%s", prop.getKey(), prop.getValue().toString()));
064      }
065    }
066    return String.format("%s{%s}", kind, String.join(", ", props));
067  }
068
069  /**
070   * Defines a property.
071   *
072   * @param name  the property name.
073   * @param value the property value.
074   * @return the same dynamic object.
075   * @throws IllegalStateException if the dynamic object is frozen.
076   */
077  public DynamicObject define(String name, Object value) {
078    frozenMutationCheck();
079    properties.put(name, value);
080    return this;
081  }
082
083  /**
084   * @return a view of all properties.
085   */
086  public Set<Map.Entry<String, Object>> properties() {
087    return properties.entrySet();
088  }
089
090  /**
091   * @param name the property name.
092   * @return the property value.
093   */
094  public Object get(String name) {
095    return properties.get(name);
096  }
097
098  /**
099   * Removes a property.
100   *
101   * @param name the property name.
102   * @return the same dynamic object.
103   */
104  public DynamicObject undefine(String name) {
105    properties.remove(name);
106    return this;
107  }
108
109  /**
110   * @return a new dynamic object whose properties point to the same objects.
111   */
112  public DynamicObject copy() {
113    DynamicObject copy = new DynamicObject(this.kind);
114    for (Map.Entry<String, Object> entry : properties.entrySet()) {
115      copy.properties.put(entry.getKey(), entry.getValue());
116    }
117    return copy;
118  }
119
120  /**
121   * Mixes all properties from another dynamic object into this one, overwriting existing properties.
122   *
123   * @param other the dynamic object to mix the properties from.
124   * @return the same dynamic object.
125   */
126  public DynamicObject mixin(DynamicObject other) {
127    frozenMutationCheck();
128    for (Map.Entry<String, Object> entry : other.properties.entrySet()) {
129      properties.put(entry.getKey(), entry.getValue());
130    }
131    return this;
132  }
133
134  /**
135   * Freezes a dynamic object, meaning that its properties cannot be added, updated and removed anymore.
136   *
137   * @return the same dynamic object.
138   */
139  public DynamicObject freeze() {
140    this.frozen = true;
141    return this;
142  }
143
144  /**
145   * Tells whether the dynamic object is frozen or not.
146   *
147   * @return {@code true} if frozen, {@code false} otherwise.
148   */
149  public boolean isFrozen() {
150    return frozen;
151  }
152
153  /**
154   * Dispatch dynamic object "methods". The receiver dynamic object is expected to be the first element of {@code args}.
155   *
156   * @param property the method property in the dynamic object.
157   * @param args     the arguments.
158   * @return the return value.
159   * @throws Throwable in case everything is wrong.
160   */
161  public static Object dispatchCall(String property, Object... args) throws Throwable {
162    DynamicObject obj = (DynamicObject) args[0];
163    Object value = obj.properties.get(property);
164    if (value != null) {
165      if (value instanceof FunctionReference) {
166        FunctionReference funRef = (FunctionReference) value;
167        if (funRef.isVarargsCollector() && args[args.length - 1] instanceof Object[]) {
168          return funRef.spread(args);
169        }
170        return funRef.invoke(args);
171      } else {
172        throw new UnsupportedOperationException("There is no dynamic object method defined for " + property);
173      }
174    }
175    if (obj.hasFallback()) {
176      FunctionReference handle = (FunctionReference) obj.properties.get("fallback");
177      Object[] fallback_args = new Object[args.length + 1];
178      fallback_args[0] = obj;
179      fallback_args[1] = property;
180      arraycopy(args, 1, fallback_args, 2, args.length - 1);
181      return handle.invoke(fallback_args);
182    }
183    throw new UnsupportedOperationException("There is neither a dynamic object method defined for " + property + " nor a 'fallback' method");
184  }
185
186  /**
187   * Dispatches getter-style dynamic object methods, i.e., methods with a receiver and no argument.
188   *
189   * @param property the method property in the dynamic object.
190   * @param object   the receiver object.
191   * @return the return value.
192   * @throws Throwable in case everything is wrong.
193   */
194  public static Object dispatchGetterStyle(String property, DynamicObject object) throws Throwable {
195    Object value = object.get(property);
196    if (value != null || object.properties.containsKey(property)) {
197      if (value instanceof FunctionReference) {
198        FunctionReference funRef = (FunctionReference) value;
199        if (funRef.acceptArity(1)) {
200          return funRef.invoke(object);
201        }
202      }
203      return value;
204    }
205    if (object.hasFallback()) {
206      FunctionReference funRef = (FunctionReference) object.properties.get("fallback");
207      return funRef.invoke(object, property);
208    }
209    return null;
210  }
211
212  /**
213   * Dispatches setter-style dynamic object methods, i.e., methods with a receiver and exactly 1 argument.
214   *
215   * @param property the method property in the dynamic object.
216   * @param object   the receiver object.
217   * @param arg      the arguments.
218   * @return the return value.
219   * @throws Throwable in case everything is wrong.
220   */
221  public static Object dispatchSetterStyle(String property, DynamicObject object, Object arg) throws Throwable {
222    Object value = object.get(property);
223    if (value != null || object.properties.containsKey(property)) {
224      if (value instanceof FunctionReference) {
225        FunctionReference funRef = (FunctionReference) value;
226        if (funRef.arity() == 2) {
227          if (funRef.isVarargsCollector() && arg instanceof Object[]) {
228            return funRef.handle().invokeExact((Object) object, (Object[]) arg);
229          }
230          return funRef.invoke(object, arg);
231        }
232      }
233    }
234    // XXX: should we try the fallback method here ?
235    return object.define(property, arg);
236  }
237
238  /**
239   * Dispatches on another dynamic object (fallback helper).
240   *
241   * @param deleguee the object to delegate to.
242   * @param receiver the receiver object.
243   * @param property the method property in the dynamic object.
244   * @param args     the arguments.
245   * @return the return value.
246   * @throws Throwable in case everything is wrong.
247   */
248  public static Object dispatchDelegate(DynamicObject deleguee, DynamicObject receiver, String property, Object... args) throws Throwable {
249    return deleguee
250      .invoker(property, genericMethodType(args.length + 1))
251      .bindTo(deleguee)
252      .invokeWithArguments(args);
253  }
254
255  /**
256   * Creates a function suitable for the {@code fallback} property delegating to the given dynamic object.
257   *
258   * Example:
259   * <pre class="listing"><code class="lang-golo" data-lang="golo">
260   * let d = DynamicObject(): name("Zaphod")
261   * let o = DynamicObject(): fallback(delegate(d))
262   * </code></pre>
263   *
264   * @param deleguee the object to delegate to.
265   * @return a function delegating to {@code deleguee}
266   */
267  public static FunctionReference delegate(DynamicObject deleguee) {
268    return new FunctionReference(
269        DISPATCH_DELEGATE.bindTo(deleguee).asVarargsCollector(Object[].class), //.asType(genericMethodType(2, true)),
270        new String[]{"this", "name", "args"});
271  }
272
273  /**
274   * Gives an invoker method handle for a given property.
275   * <p>
276   * While this method may be useful in itself, it is mostly relevant for the Golo runtime internals so as
277   * to allow calling "methods" on dynamic objects, as in:
278   * <pre class="listing"><code class="lang-golo" data-lang="golo">
279   * # obj is some dynamic object...
280   * obj: foo("bar")
281   * println(foo: bar())
282   *
283   * obj: define("plop", |this| -> "Plop!")
284   * println(obj: plop())
285   * </code></pre>
286   *
287   * @param property the name of a property.
288   * @param type     the expected invoker type with at least one parameter (the dynamic object as a receiver).
289   * @return a method handle.
290   */
291  public MethodHandle invoker(String property, MethodType type) {
292    switch (type.parameterCount()) {
293      case 0:
294        throw new IllegalArgumentException("A dynamic object invoker type needs at least 1 argument (the receiver)");
295      case 1:
296        return DISPATCH_GET.bindTo(property).asType(genericMethodType(1));
297      case 2:
298        return DISPATCH_SET.bindTo(property).asType(genericMethodType(2));
299      default:
300        return DISPATCH_CALL.bindTo(property).asCollector(Object[].class, type.parameterCount());
301    }
302  }
303
304  /**
305   * Verify if a method is defined for the dynamic object.
306   *
307   * @param method the method name.
308   * @return {@code true} if method is defined, {@code false} otherwise.
309   */
310  public boolean hasMethod(String method) {
311    Object obj = properties.get(method);
312    if (obj != null) {
313      return isClosure(obj);
314    }
315    return false;
316  }
317
318  /**
319   * Let the user define a fallback behavior.
320   *
321   * @param value the fallback value
322   * @return the current object
323   */
324  public DynamicObject fallback(Object value) {
325    return define("fallback", value);
326  }
327
328  /**
329   * Verify a fallback property exists.
330   *
331   * @return {@code true} if a fallback behavior is defined, {@code false} otherwise.
332   */
333  private boolean hasFallback() {
334    return properties.containsKey("fallback");
335  }
336
337  public static final MethodHandle DISPATCH_CALL;
338  public static final MethodHandle DISPATCH_GET;
339  public static final MethodHandle DISPATCH_SET;
340  public static final MethodHandle DISPATCH_DELEGATE;
341
342  static {
343    MethodHandles.Lookup lookup = MethodHandles.lookup();
344    try {
345      DISPATCH_DELEGATE = lookup.findStatic(DynamicObject.class, "dispatchDelegate",
346          methodType(Object.class, DynamicObject.class, DynamicObject.class, String.class, Object[].class));
347      DISPATCH_CALL = lookup.findStatic(DynamicObject.class, "dispatchCall", methodType(Object.class, String.class, Object[].class));
348      DISPATCH_GET = lookup.findStatic(DynamicObject.class, "dispatchGetterStyle", methodType(Object.class, String.class, DynamicObject.class));
349      DISPATCH_SET = lookup.findStatic(DynamicObject.class, "dispatchSetterStyle", methodType(Object.class, String.class, DynamicObject.class, Object.class));
350    } catch (NoSuchMethodException | IllegalAccessException e) {
351      e.printStackTrace();
352      throw new Error("Could not bootstrap the required method handles");
353    }
354  }
355
356  private void frozenMutationCheck() {
357    if (frozen) {
358      throw new IllegalStateException("the object is frozen");
359    }
360  }
