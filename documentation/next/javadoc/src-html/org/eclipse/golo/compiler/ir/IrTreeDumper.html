011package org.eclipse.golo.compiler.ir;
012
013public class IrTreeDumper implements GoloIrVisitor {
014
015  private int spacing = 0;
016
017  private void space() {
018    System.out.print("# ");
019    for (int i = 0; i < spacing; i++) {
020      System.out.print(" ");
021    }
022  }
023
024  private void incr() {
025    spacing = spacing + 2;
026  }
027
028  private void decr() {
029    spacing = spacing - 2;
030  }
031
032  @Override
033  public void visitModule(GoloModule module) {
034    space();
035    System.out.println(module.getPackageAndClass());
036    module.walk(this);
037  }

057  @Override
058  public void visitAugmentation(Augmentation augmentation) {
059    incr();
060    space();
061    System.out.println("Augmentation on " + augmentation.getTarget());
062    if (augmentation.hasNames()) {
063      incr();

071    decr();
072  }

123  private void visitFunctionDefinition(GoloFunction function) {
124    System.out.print("|");
125    boolean first = true;
126    for (String param : function.getParameterNames()) {
127      if (first) {
128        first = false;
129      } else {
130        System.out.print(", ");
131      }
132      System.out.print(param);
133    }
134    System.out.print("|");
135    if (function.isVarargs()) {
136      System.out.print(" (varargs)");
137    }
138    if (function.isSynthetic()) {
139      System.out.print(" (synthetic, " + function.getSyntheticParameterCount() + " synthetic parameters)");
140      if (function.getSyntheticSelfName() != null) {
141        System.out.print(" (selfname: " + function.getSyntheticSelfName() + ")");
142      }
143    }
144    System.out.println();
145    function.walk(this);
146  }

316  @Override
317  public void visitForEachLoopStatement(ForEachLoopStatement foreachStatement) {
318    incr();
319    space();
320    System.out.println("Foreach");
321    incr();
322    for (LocalReference ref : foreachStatement.getReferences()) {
323      ref.accept(this);
324    }
325    foreachStatement.getIterable().accept(this);
326    if (foreachStatement.hasWhenClause()) {
327      space();
328      System.out.println("When:");
329      foreachStatement.getWhenClause().accept(this);
330    }
331    foreachStatement.getBlock().accept(this);
332    decr();
333    decr();
334  }

357  @Override
358  public void visitTryCatchFinally(TryCatchFinally tryCatchFinally) {
359    incr();
360    space();
361    System.out.println("Try");
362    tryCatchFinally.getTryBlock().accept(this);
363    if (tryCatchFinally.hasCatchBlock()) {
364      space();
365      System.out.println("Catch: " + tryCatchFinally.getExceptionId());
366      tryCatchFinally.getCatchBlock().accept(this);
367    }
368    if (tryCatchFinally.hasFinallyBlock()) {
369      space();
370      System.out.println("Finally");
371      tryCatchFinally.getFinallyBlock().accept(this);
372    }
373    decr();
374  }

376  @Override
377  public void visitClosureReference(ClosureReference closureReference) {
378    GoloFunction target = closureReference.getTarget();
379    incr();
380    space();
381    if (target.isAnonymous()) {
382      System.out.print("Closure: ");
383      incr();
384      visitFunctionDefinition(target);
385      decr();
386    } else {
387      System.out.printf(
388          "Closure reference: %s, regular arguments at index %d%n",
389          target.getName(),
390          target.getSyntheticParameterCount());
391      incr();
392      for (String refName : closureReference.getCapturedReferenceNames()) {
393        space();
394        System.out.println("- capture: " + refName);
395      }
396      decr();
397    }
398    decr();
399  }

455  private void printLocalDeclarations(ExpressionStatement expr) {
456    if (expr.hasLocalDeclarations()) {
457      incr();
458      space();
459      System.out.println("Local declaration:");
460      for (GoloAssignment a : expr.declarations()) {
461        a.accept(this);
462      }
463      decr();
464    }
