011package org.eclipse.golo.compiler.ir;
012
013import java.util.List;
014import java.util.LinkedList;
015
016import org.eclipse.golo.compiler.parser.GoloASTNode;
017
018import static java.util.Collections.unmodifiableList;
019
020public final class MatchExpression extends ExpressionStatement implements Alternatives<ExpressionStatement> {
021
022  private ExpressionStatement otherwise;
023  private final LinkedList<WhenClause<ExpressionStatement>> clauses = new LinkedList<>();
024
025  MatchExpression() {
026    super();
027  }
028
029  public MatchExpression when(Object cond) {
030    WhenClause<ExpressionStatement> clause = new WhenClause<ExpressionStatement>(ExpressionStatement.of(cond), null);
031    this.clauses.add(clause);
032    makeParentOf(clause);
033    return this;
034  }
035
036  public MatchExpression then(Object action) {
037    this.clauses.getLast().setAction((ExpressionStatement) action);
038    return this;
039  }
040
041  public MatchExpression otherwise(Object action) {
042    otherwise = (ExpressionStatement) action;
043    makeParentOf(otherwise);
044    return this;
045  }
046
047  public List<WhenClause<ExpressionStatement>> getClauses() {
048    return unmodifiableList(this.clauses);
049  }
050
051  public ExpressionStatement getOtherwise() {
052    return this.otherwise;
053  }
054
055  @Override
056  public MatchExpression ofAST(GoloASTNode node) {
057    super.ofAST(node);
058    return this;
059  }
060
061  @Override
062  public void accept(GoloIrVisitor visitor) {
063    visitor.visitMatchExpression(this);
064  }
065
066  @Override
067  public void walk(GoloIrVisitor visitor) {
068    for (WhenClause<ExpressionStatement> clause : clauses) {
069      clause.accept(visitor);
070    }
071    otherwise.accept(visitor);
072  }
073
074  @Override
075  protected void replaceElement(GoloElement original, GoloElement newElement) {
076    if (!(newElement instanceof ExpressionStatement || newElement instanceof WhenClause)) {
077      throw cantConvert("ExpressionStatement or WhenClause", newElement);
078    }
079    if (otherwise.equals(original)) {
080      otherwise(newElement);
081      return;
082    }
083    if (clauses.contains(original)) {
084      @SuppressWarnings("unchecked")
085      WhenClause<ExpressionStatement> when = (WhenClause<ExpressionStatement>) newElement;
086      clauses.set(clauses.indexOf(original), when);
087      makeParentOf(when);
088      return;
089    }
090    throw doesNotContain(original);
091  }
092}
