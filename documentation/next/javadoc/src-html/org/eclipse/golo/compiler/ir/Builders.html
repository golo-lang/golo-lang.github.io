152  public static ConstantStatement constant(Object value) {
153    if (value instanceof Class) {
154      return classRef(value);
155    }
156    if (value instanceof ConstantStatement) {
157      return (ConstantStatement) value;
158    }
159    return new ConstantStatement(value);
160  }
161
162  public static ConstantStatement classRef(Object cls) {
163    if (cls instanceof String) {
164      return constant(toClassRef((String) cls));
165    }
166    if (cls instanceof Class) {
167      return constant(toClassRef((Class) cls));
168    }
169    if (cls instanceof GoloParser.ParserClassRef) {
170      return constant(cls);
171    }
172    if (cls instanceof PackageAndClass) {
173      return constant(toClassRef(cls.toString()));
174    }
175    throw new IllegalArgumentException("unknown type " + cls.getClass() + "to build a class reference");
176  }

302  public static GoloStatement toGoloStatement(Object statement) {
303    if (statement == null) { return null; }
304    if (statement instanceof GoloStatement) {
305      return (GoloStatement) statement;
306    }
307    throw cantConvert(statement, "GoloStatement");
308  }
309
310  public static Block toBlock(Object block) {
311    if (block == null) { return Block.emptyBlock(); }
312    if (block instanceof Block) {
313      return (Block) block;
314    }
315    throw cantConvert(block, "Block");
316  }
317
318  private static IllegalArgumentException cantConvert(Object value, String target) {
319    return new IllegalArgumentException(
320        String.format("%s is not a %s, but a %s",
321        value, target, value.getClass()));
322  }

332  public static ModuleImport moduleImport(Object name) {
333    if (name instanceof String) {
334      return new ModuleImport(PackageAndClass.fromString((String) name));
335    }
336    if (name instanceof PackageAndClass) {
337      return new ModuleImport((PackageAndClass) name);
338    }
339    throw cantConvert(name, "string or package");
340  }

346  public static ConditionalBranching branch(Object condition,
347                                            Block trueBlock,
348                                            Block falseBlock,
349                                            ConditionalBranching elseBranch) {
350    return branch().condition(condition)
351      .whenTrue(trueBlock)
352      .whenFalse(falseBlock)
353      .elseBranch(elseBranch);
