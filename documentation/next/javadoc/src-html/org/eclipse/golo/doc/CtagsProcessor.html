011package org.eclipse.golo.doc;
012
013import org.eclipse.golo.compiler.parser.ASTCompilationUnit;
014import gololang.IO;
015
016import java.nio.file.Path;
017import java.util.Map;
018import java.util.LinkedList;
019
020public class CtagsProcessor extends AbstractProcessor {
021
022  private final LinkedList<String> ctags = new LinkedList<>();
023  private String file = "file";
024
025  private void ctagsLine(String name, String address, String field) {
026    ctags.add(String.format("%s\t%s\t%s;\"\t%s\tlanguage:golo%n", name, file, address, field));
027  }
028
029  private void ctagsModule(ModuleDocumentation module) {
030    ctagsLine(module.moduleName(),
031        "/^module[:blank:]+" + module.moduleName().replace(".", "\\.") + "$/",
032        "p\tline:" + module.moduleDefLine());
033  }
034
035  private void ctagsFunction(FunctionDocumentation funct) {
036    ctagsFunction(funct, "", false);
037  }
038
039  private void ctagsFunction(FunctionDocumentation funct, String parent, boolean named) {
040    String address = String.format("/function[:blank:]+%s[:blank:]+=/", funct.name());
041
042    StringBuilder signature = new StringBuilder("\tsignature:(");
043    if (funct.arity() > 0) {
044      signature.append(funct.argument(0));
045      for (int i = 1; i < funct.arity(); i++) {
046        signature.append(", ").append(funct.argument(i));
047      }
048      if (funct.varargs()) { signature.append("..."); }
049    }
050    signature.append(")");
051
052    StringBuilder fields = new StringBuilder("f");
053    fields.append("\tline:").append(funct.line());
054    if (funct.local()) {
055      fields.append("\taccess:private\tfile:");
056    } else {
057      fields.append("\taccess:public");
058    }
059    fields.append(signature);
060    if (!parent.isEmpty()) {
061      if (named) {
062        fields.append("\taugmentation:").append(parent);
063      } else {
064        fields.append("\taugment:").append(parent);
065      }
066    }
067    ctagsLine(funct.name(), address, fields.toString());
068  }
069
070  private void ctagsAugment(String name, int line) {
071    ctagsLine(name,
072        String.format("/^augment[:blank:]+%s/", name.replace(".", "\\.")),
073        String.format("a\tline:%s", line));
074  }
075
076  private void ctagsAugmentation(String name, int line) {
077    ctagsLine(name,
078        String.format("/^augmentation[:blank:]+%s[:blank:]+=[:blank:]+{/", name),
079        String.format("na\tline:%s", line));
080  }
081
082  private void ctagsStruct(String name, int line) {
083    ctagsLine(name,
084        String.format("/^struct[:blank:]+%s[:blank:]+=/", name),
085        String.format("s\tline:%s", line));
086  }
087
088  private void ctagsUnion(UnionDocumentation unionDoc) {
089    ctagsLine(unionDoc.name(),
090        String.format("/^union[:blank:]+%s[:blank:]+=[:blank:]+{/", unionDoc.name()),
091        String.format("g\tline:%s", unionDoc.line()));
092
093    for (UnionDocumentation.UnionValueDocumentation valueDoc : unionDoc.values()) {
094      ctagsUnionValue(unionDoc.name(), valueDoc);
095    }
096  }
097
098  private void ctagsUnionValue(String unionName, UnionDocumentation.UnionValueDocumentation valueDoc) {
099    ctagsLine(valueDoc.name(),
100        String.format("/[:blank:]+%s[:blank:]+%s", valueDoc.name(),
101                    valueDoc.hasMembers() ? "[:blank:]*=[:blank:]+{" : ""),
102        String.format("e\tline:%s\tunion:%s", valueDoc.line(), unionName));
103
104    for (MemberDocumentation member : valueDoc.members()) {
105      ctagsLine(member.name(),
106        String.format("/[:blank:]+%s[:blank:]+=/", valueDoc.name()),
107        String.format("m\tline:%s\taccess:public\tvalue:%s",
108          member.line(),
109          valueDoc.name()));
110    }
111  }
112
113  private void ctagsImport(String name, int line) {
114    ctagsLine(name,
115        String.format("/^import[:blank:]+%s/", name.replace(".", "\\.")),
116        String.format("i\tline:%s", line));
117  }
118
119  private void ctagsModState(String name, int line) {
120    ctagsLine(name,
121        String.format("(let|var)[:blank:]+%s[:blank:]+=/", name),
122        String.format("v\taccess:private\tfile:\tline:%s", line));
123  }
124
125  private void ctagsStructMember(String struct, String member, int line) {
126    ctagsLine(member,
127        String.format("/struct[:blank:]+%s[:blank:]+=/", struct),
128        String.format("m\tline:%s\taccess:%s\tstruct:%s",
129          line,
130          (member.charAt(0) == '_') ? "private" : "public",
131          struct));
132  }
133
134  private String ctagsAsString() {
135    java.util.Collections.sort(ctags);
136    StringBuilder buffer = new StringBuilder();
137    for (String line : ctags) {
138      buffer.append(line);
139    }
140    return buffer.toString();
141  }
142
143  @Override
144  public String render(ModuleDocumentation documentation) throws Throwable {
145    ctagsModule(documentation);
146    for (Map.Entry<String, Integer> imp : documentation.imports().entrySet()) {
147      ctagsImport(imp.getKey(), imp.getValue());
148    }
149    for (StructDocumentation struct : documentation.structs()) {
150      ctagsStruct(struct.name(), struct.line());
151      for (MemberDocumentation member : struct.members()) {
152        ctagsStructMember(struct.name(), member.name(), member.line());
153      }
154    }
155    for (UnionDocumentation unionDoc : documentation.unions()) {
156      ctagsUnion(unionDoc);
157    }
158    for (NamedAugmentationDocumentation augment : documentation.namedAugmentations()) {
159      ctagsAugmentation(augment.name(), augment.line());
160      for (FunctionDocumentation funct : augment.functions()) {
161        ctagsFunction(funct, augment.name(), true);
162      }
163    }
164    for (AugmentationDocumentation augment : documentation.augmentations()) {
165      ctagsAugment(augment.target(), augment.line());
166      for (FunctionDocumentation funct : augment.functions()) {
167        ctagsFunction(funct, augment.target(), false);
168      }
169    }
170    for (Map.Entry<String, Integer> state : documentation.moduleStates().entrySet()) {
171      ctagsModState(state.getKey(), state.getValue());
172    }
173    for (FunctionDocumentation funct : documentation.functions(true)) {
174      ctagsFunction(funct);
175    }
176    return ctagsAsString();
177  }
178
179  @Override
180  public void process(Map<String, ModuleDocumentation> modules, Path targetFolder) throws Throwable {
181    Path targetFile = null;
182    if (targetFolder.toString().equals("-")) {
183      targetFile = targetFolder;
184    } else {
185      targetFile = targetFolder.resolve("tags");
186    }
187    ctags.clear();
188    for (Map.Entry<String, ModuleDocumentation> src : modules.entrySet()) {
189      file = src.getKey();
190      render(src.getValue());
191    }
192    IO.textToFile(ctagsAsString(), targetFile);
193  }
194}
