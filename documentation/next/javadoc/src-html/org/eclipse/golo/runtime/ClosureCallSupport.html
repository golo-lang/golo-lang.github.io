011package org.eclipse.golo.runtime;
012
013import gololang.FunctionReference;
014
015import java.lang.invoke.*;
016import java.util.Arrays;
017
018import static java.lang.invoke.MethodHandles.guardWithTest;
019import static java.lang.invoke.MethodHandles.permuteArguments;
020import static java.lang.invoke.MethodType.methodType;
021import static org.eclipse.golo.runtime.NamedArgumentsHelper.checkArgumentPosition;
022
023public final class ClosureCallSupport {
024
025  private ClosureCallSupport() {
026    throw new UnsupportedOperationException("Don't instantiate invokedynamic bootstrap class");
027  }
028
029  static class InlineCache extends MutableCallSite {
030
031    MethodHandle fallback;
032    final boolean constant;
033    final String[] argumentNames;
034
035    InlineCache(MethodType type, boolean constant, String[] argumentNames) {
036      super(type);
037      this.constant = constant;
038      this.argumentNames = argumentNames;
039    }
040  }
041
042  private static final MethodHandle GUARD;
043  private static final MethodHandle FALLBACK;
044
045  static {
046    try {
047      MethodHandles.Lookup lookup = MethodHandles.lookup();
048
049      GUARD = lookup.findStatic(
050          ClosureCallSupport.class,
051          "guard",
052          methodType(boolean.class, FunctionReference.class, FunctionReference.class));
053
054      FALLBACK = lookup.findStatic(
055          ClosureCallSupport.class,
056          "fallback",
057          methodType(Object.class, InlineCache.class, Object[].class));
058    } catch (NoSuchMethodException | IllegalAccessException e) {
059      throw new Error("Could not bootstrap the required method handles", e);
060    }
061  }
062
063  public static CallSite bootstrap(MethodHandles.Lookup caller, String name, MethodType type, Object... bsmArgs) {
064    boolean constant = ((int) bsmArgs[0]) == 1;
065    String[] argumentNames = new String[bsmArgs.length - 1];
066    for (int i = 0; i < bsmArgs.length - 1; i++) {
067      argumentNames[i] = (String) bsmArgs[i + 1];
068    }
069    InlineCache callSite = new InlineCache(type, constant, argumentNames);
070    MethodHandle fallbackHandle = FALLBACK
071        .bindTo(callSite)
072        .asCollector(Object[].class, type.parameterCount())
073        .asType(type);
074    callSite.fallback = fallbackHandle;
075    callSite.setTarget(fallbackHandle);
076    return callSite;
077  }
078
079  public static boolean guard(FunctionReference expected, FunctionReference actual) {
080    return expected == actual;
081  }
082
083  public static Object fallback(InlineCache callSite, Object[] args) throws Throwable {
084    FunctionReference targetFunctionReference = (FunctionReference) args[0];
085    MethodHandle target = targetFunctionReference.handle();
086    MethodHandle invoker = MethodHandles.dropArguments(target, 0, FunctionReference.class);
087    MethodType type = invoker.type();
088    if (callSite.argumentNames.length > 0) {
089      invoker = reorderArguments(
090          targetFunctionReference.parameterNames(),
091          invoker,
092          callSite.argumentNames);
093    }
094    if (target.isVarargsCollector()) {
095      if (TypeMatching.isLastArgumentAnArray(type.parameterCount(), args)) {
096        invoker = invoker.asFixedArity().asType(callSite.type());
097      } else {
098        invoker = invoker.asCollector(
099            Object[].class,
100            callSite.type().parameterCount() - target.type().parameterCount())
101          .asType(callSite.type());
102      }
103    } else {
104      invoker = invoker.asType(callSite.type());
105    }
106    if (callSite.constant) {
107      Object constantValue = invoker.invokeWithArguments(args);
108      MethodHandle constant;
109      if (constantValue == null) {
110        constant = MethodHandles.constant(Object.class, null);
111      } else {
112        constant = MethodHandles.constant(constantValue.getClass(), constantValue);
113      }
114      constant = MethodHandles.dropArguments(constant, 0,  type.parameterArray());
115      callSite.setTarget(constant.asType(type));
116      return constantValue;
117    } else {
118      MethodHandle guard = GUARD.bindTo(targetFunctionReference);
119      MethodHandle root = guardWithTest(guard, invoker, callSite.fallback);
120      callSite.setTarget(root);
121      return invoker.invokeWithArguments(args);
122    }
123  }
124
125  private static MethodHandle reorderArguments(String[] parameterNames, MethodHandle handle, String[] argumentNames) {
126    if (parameterNames.length > 0) {
127      int[] argumentsOrder = new int[parameterNames.length + 1];
128      argumentsOrder[0] = 0;
129      argumentsOrder[1] = 1;
130      for (int i = 0; i < argumentNames.length; i++) {
131        int actualPosition = -1;
132        for (int j = 0; j < parameterNames.length; j++) {
133          if (parameterNames[j].equals(argumentNames[i])) {
134            actualPosition = j;
135          }
136        }
137        checkArgumentPosition(actualPosition, argumentNames[i], "closure " + Arrays.toString(parameterNames));
138        argumentsOrder[actualPosition + 1] = i + 1;
139      }
140      return permuteArguments(handle, handle.type(), argumentsOrder);
141    }
142    Warnings.noParameterNames("closure " + Arrays.toString(parameterNames), argumentNames);
143    return handle;
144  }
145}
