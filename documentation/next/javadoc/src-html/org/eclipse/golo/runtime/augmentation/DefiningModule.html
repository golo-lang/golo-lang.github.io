011package org.eclipse.golo.runtime.augmentation;
012
013import java.util.stream.Stream;
014import java.util.function.Predicate;
015import org.eclipse.golo.runtime.Loader;
016import org.eclipse.golo.runtime.Module;
017
018import static org.eclipse.golo.runtime.augmentation.AugmentationApplication.Kind;
019
020/**
021 * Encapsulate a module defining an augmentation.
022 */
023public final class DefiningModule {
024
025  public enum Scope { LOCAL, IMPORT, CALLSTACK }
026
027  private final Class<?> module;
028  private final Scope scope;
029
030  DefiningModule(Class<?> module, Scope scope) {
031    this.module = module;
032    this.scope = scope;
033  }
034
035  public Class<?> module() {
036    return this.module;
037  }
038
039  public static DefiningModule of(Class<?> module, Scope scope) {
040    return new DefiningModule(module, scope);
041  }
042
043  public static DefiningModule ofLocal(Class<?> module) {
044    return new DefiningModule(module, Scope.LOCAL);
045  }
046
047  public static DefiningModule ofImport(Class<?> module) {
048    return new DefiningModule(module, Scope.IMPORT);
049  }
050
051  public static DefiningModule ofCallstack(Class<?> module) {
052    return new DefiningModule(module, Scope.CALLSTACK);
053  }
054
055  /**
056   * Returns a stream of augmentations definitions defined directly on a type in the corresponding
057   * module.
058   * e.g.
059   * <pre class="listing"><code class="lang-golo" data-lang="golo">
060   * augment module.Type {
061   *    # ...
062   * }
063   * </code></pre>
064   */
065  private Stream<AugmentationApplication> simpleAugmentationsFor(Loader loader, Class<?> receiverType) {
066    return Stream.of(Module.augmentations(module))
067      .map(loader)
068      .filter(isAssignableFrom(receiverType))
069      .map(target -> new AugmentationApplication(
070            loader.load(module.getName() + "$" + target.getName().replace(".", "$")),
071            target, scope, Kind.SIMPLE));
072  }
073
074  private static Predicate<Class<?>> isAssignableFrom(Class<?> receiver) {
075    return target -> target != null && target.isAssignableFrom(receiver);
076  }
077
078  private Stream<AugmentationApplication> fullyNamedAugmentationsFor(Loader loader, Class<?> receiverType) {
079    return Stream.of(Module.augmentationApplications(module))
080      .map(targetName -> loader.load(targetName))
081      .filter(target -> target != null && target.isAssignableFrom(receiverType))
082      .flatMap(target -> qualifyAugmentations(loader, target));
083  }
084
085  private Stream<AugmentationApplication> qualifyAugmentations(Loader loader, Class<?> target) {
086    return Stream.of(Module.augmentationApplications(module, target))
087      .flatMap(augmentName -> fullyQualifiedName(augmentName))
088      .map(augmentName -> new AugmentationApplication(
089              loader.load(augmentName),
090              target, scope, Kind.NAMED));
091  }
092
093  /**
094   * Fully qualify an augmentation name.
095   * <p>
096   * Given an augmentation name, this generate a stream of alternative names by prepending
097   * names of the defining module as well as imported modules.
098   */
099  private Stream<String> fullyQualifiedName(String augmentationName) {
100    Stream.Builder<String> names = Stream.builder();
101    int idx = augmentationName.lastIndexOf(".");
102    if (idx == -1) {
103      names.add(augmentationName);
104    } else {
105      names.add(new StringBuilder(augmentationName).replace(idx, idx + 1, "$").toString());
106    }
107    names.add(module.getName() + "$" + augmentationName.replace(".", "$"));
108    return Stream.concat(
109        names.build(),
110        Stream.of(Module.imports(module))
111        .map(prefix -> prefix + "$" + augmentationName.replace(".", "$")));
112  }
113
114  public Stream<AugmentationApplication> augmentationsFor(Loader loader, Class<?> receiverType) {
115    if (module == null) {
116      return Stream.empty();
117    }
118    return Stream.concat(
119        simpleAugmentationsFor(loader, receiverType),
120        fullyNamedAugmentationsFor(loader, receiverType));
121
122  }
123
124  @Override
125  public String toString() {
126    return "DefiningModule<" + module + "," + scope + ">";
127  }
128}
