011package org.eclipse.golo.runtime;
012
013import gololang.FunctionReference;
014
015import java.lang.invoke.*;
016import java.lang.reflect.Method;
017import java.lang.reflect.Parameter;
018
019import static java.lang.invoke.MethodHandles.constant;
020import static java.lang.invoke.MethodType.genericMethodType;
021
022public final class ClosureReferenceSupport {
023
024  private ClosureReferenceSupport() {
025    throw new UnsupportedOperationException("Don't instantiate invokedynamic bootstrap class");
026  }
027
028  public static CallSite bootstrap(MethodHandles.Lookup caller, String name, MethodType type, String moduleClass, int arity, int varargs) throws Throwable {
029    Class<?> module = caller.lookupClass().getClassLoader().loadClass(moduleClass);
030    Method function = module.getDeclaredMethod(name, genericMethodType(arity, varargs == 1).parameterArray());
031    function.setAccessible(true);
032    return new ConstantCallSite(constant(
033          FunctionReference.class,
034          new FunctionReference(caller.unreflect(function), parameterNames(function))));
035  }
036
037  private static String[] parameterNames(Method function) {
038    Parameter[] parameters = function.getParameters();
039    String[] parameterNames = new String[parameters.length];
040    for (int i = 0; i < parameters.length; i++) {
041      parameterNames[i] = parameters[i].getName();
042    }
043    return parameterNames;
044  }
045
046}
