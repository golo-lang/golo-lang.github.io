035  static final class InlineCache extends MutableCallSite {
036
037    static final int MEGAMORPHIC_THRESHOLD = 5;
038
039    final MethodHandles.Lookup callerLookup;
040    final String name;
041    final boolean nullSafeGuarded;
042    final String[] argumentNames;
043
044    int depth = 0;
045    MethodHandle resetFallback;
046    WeakHashMap<Class<?>, MethodHandle> vtable;
047
048    InlineCache(Lookup callerLookup, String name, MethodType type, boolean nullSafeGuarded, String... argumentNames) {
049      super(type);
050      this.callerLookup = callerLookup;
051      this.name = name;
052      this.nullSafeGuarded = nullSafeGuarded;
053      this.argumentNames = argumentNames;
054    }
055
056    boolean isMegaMorphic() {
057      return depth > MEGAMORPHIC_THRESHOLD;
058    }
059
060    public MethodInvocation toMethodInvocation(Object[] args) {
061      return new MethodInvocation(name, type(), args, argumentNames);
062    }
063  }

148  public static CallSite bootstrap(Lookup caller, String name, MethodType type, Object... bsmArgs) {
149    boolean nullSafeGuarded = ((int) bsmArgs[0]) == 1;
150    String[] argumentNames = new String[bsmArgs.length - 1];
151    for (int i = 0; i < bsmArgs.length - 1; i++) {
152      argumentNames[i] = (String) bsmArgs[i + 1];
153    }
154    InlineCache callSite = new InlineCache(caller, name, type, nullSafeGuarded, argumentNames);
155    MethodHandle fallbackHandle = FALLBACK
156        .bindTo(callSite)
157        .asCollector(Object[].class, type.parameterCount())
158        .asType(type);
159    callSite.resetFallback = RESET_FALLBACK
160        .bindTo(callSite)
161        .asCollector(Object[].class, type.parameterCount())
162        .asType(type);
163    callSite.setTarget(fallbackHandle);
164    return callSite;
165  }

167  public static boolean classGuard(Class<?> expected, Object receiver) {
168    return receiver.getClass() == expected;
169  }
170
171  public static boolean overloadedGuard_generic(Class<?>[] types, Object[] arguments) {
172    if (arguments[0].getClass() != types[0]) {
173      return false;
174    }
175    for (int i = 1; i < types.length; i++) {
176      if ((arguments[i] != null) && (arguments[i].getClass() != types[i])) {
177        return false;
178      }
179    }
180    return true;
181  }

184    return receiver.getClass() == t1 &&
185        (arg == null || arg.getClass() == t2);

209  public static MethodHandle vtableLookup(InlineCache inlineCache, Object[] args) {
210    Class<?> receiverClass = args[0].getClass();
211    MethodHandle target = inlineCache.vtable.get(receiverClass);
212    if (target == null) {
213      target = lookupTarget(receiverClass, inlineCache, args);
214      inlineCache.vtable.put(receiverClass, target);
215    }
216    return target;
217  }
218
219  private static MethodHandle lookupTarget(Class<?> receiverClass, InlineCache inlineCache, Object[] args) {
220    MethodInvocation invocation = inlineCache.toMethodInvocation(args);
221    if (receiverClass.isArray()) {
222      return new ArrayMethodFinder(invocation, inlineCache.callerLookup).find();
223    }
224    if (isCallOnDynamicObject(inlineCache, args[0])) {
225      DynamicObject dynamicObject = (DynamicObject) args[0];
226      return dynamicObject.invoker(inlineCache.name, inlineCache.type());
227    } else {
228      return findTarget(invocation, inlineCache);
229    }
230  }

232  public static Object resetFallback(InlineCache inlineCache, Object[] args) throws Throwable {
233    inlineCache.depth = 0;
234    return fallback(inlineCache, args);
235  }

237  public static Object fallback(InlineCache inlineCache, Object[] args) throws Throwable {
238
239    if (inlineCache.isMegaMorphic()) {
240      return installVTableDispatch(inlineCache, args);
241    }
242
243    if (args[0] == null) {
244      if (shouldReturnNull(inlineCache, args[0])) {
245        return null;
246      } else {
247        throw new NullPointerException("On method: "
248            + inlineCache.name + " " + inlineCache.type().dropParameterTypes(0, 1));
249      }
250    }
251
252    Class<?> receiverClass = args[0].getClass();
253    MethodHandle target = lookupTarget(receiverClass, inlineCache, args);
254
255    if (target == null) {
256      // TODO: extract method to look for a `fallback` method on the receiver
257      InlineCache fallbackCallSite = new InlineCache(
258          inlineCache.callerLookup,
259          "fallback",
260          methodType(Object.class, Object.class, Object.class, Object[].class),
261          false);
262      Object[] fallbackArgs = new Object[]{
263          args[0],
264          inlineCache.name,
265          Arrays.copyOfRange(args, 1, args.length)
266      };
267      target = lookupTarget(receiverClass, fallbackCallSite, fallbackArgs);
268      if (target != null) {
269        return fallback(fallbackCallSite, fallbackArgs);
270      } else {
271        throw new NoSuchMethodError(receiverClass + "::" + inlineCache.name);
272      }
273    }
274
275    MethodHandle guard = CLASS_GUARD.bindTo(receiverClass);
276    MethodHandle fallback = inlineCache.getTarget();
277    MethodHandle root = guardWithTest(guard, target, fallback);
278    if (inlineCache.nullSafeGuarded) {
279      root = makeNullSafeGuarded(root);
280    }
281    inlineCache.setTarget(root);
282    inlineCache.depth = inlineCache.depth + 1;
283    return target.invokeWithArguments(args);
284  }
285
286  private static MethodHandle makeNullSafeGuarded(MethodHandle root) {
287    MethodHandle catchThenNull = dropArguments(constant(Object.class, null), 0, NullPointerException.class);
288    root = catchException(root, NullPointerException.class, catchThenNull);
289    return root;
290  }
291
292  private static boolean shouldReturnNull(InlineCache inlineCache, Object arg) {
293    return (arg == null) && inlineCache.nullSafeGuarded;
294  }
295
296  private static Object installVTableDispatch(InlineCache inlineCache, Object[] args) throws Throwable {
297    if (inlineCache.vtable == null) {
298      inlineCache.vtable = new WeakHashMap<>();
299    }
300    MethodHandle lookup = VTABLE_LOOKUP
301        .bindTo(inlineCache)
302        .asCollector(Object[].class, args.length);
303    MethodHandle exactInvoker = exactInvoker(inlineCache.type());
304    MethodHandle vtableTarget = foldArguments(exactInvoker, lookup);
305    if (inlineCache.nullSafeGuarded) {
306      vtableTarget = makeNullSafeGuarded(vtableTarget);
307    }
308    inlineCache.setTarget(vtableTarget);
309    if (shouldReturnNull(inlineCache, args[0])) {
310      return null;
311    }
312    return vtableTarget.invokeWithArguments(args);
313  }

315  private static boolean isCallOnDynamicObject(InlineCache inlineCache, Object arg) {
316    return (arg instanceof DynamicObject)
317      && !DYNAMIC_OBJECT_RESERVED_METHOD_NAMES.contains(inlineCache.name)
318      && (!"toString".equals(inlineCache.name) || ((DynamicObject) arg).hasMethod("toString"));
319  }

321  private static MethodHandle guardOnOverloaded(MethodHandle target, MethodInvocation invocation, MethodHandle reset) {
322    Object[] args = invocation.arguments();
323    Class<?>[] types = new Class<?>[args.length];
324    for (int i = 0; i < types.length; i++) {
325      types[i] = (args[i] == null) ? Object.class : args[i].getClass();
326    }
327    MethodHandle guard;
328    switch (args.length) {
329      case 2:
330        guard = insertArguments(OVERLOADED_GUARD_1, 0, types[0], types[1]);
331        break;
332      case 3:
333        guard = insertArguments(OVERLOADED_GUARD_2, 0, types[0], types[1], types[2]);
334        break;
335      case 4:
336        guard = insertArguments(OVERLOADED_GUARD_3, 0, types[0], types[1], types[2], types[3]);
337        break;
338      case 5:
339        guard = insertArguments(OVERLOADED_GUARD_4, 0, types[0], types[1], types[2], types[3], types[4]);
340        break;
341      default:
342        guard = OVERLOADED_GUARD_GENERIC.bindTo(types).asCollector(Object[].class, types.length);
343    }
344    return guardWithTest(guard, target, reset);
345  }

347  private static MethodHandle findTarget(MethodInvocation invocation, InlineCache inlineCache) {
348    MethodHandle target;
349    Lookup lookup = inlineCache.callerLookup;
350
351    RegularMethodFinder regularMethodFinder = new RegularMethodFinder(invocation, lookup);
352    target = regularMethodFinder.find();
353    if (target != null) {
354      if (regularMethodFinder.isOverloaded()) {
355        return guardOnOverloaded(target, invocation, inlineCache.resetFallback);
356      }
357      return target;
358    }
359
360    target = new PropertyMethodFinder(invocation, lookup).find();
361    if (target != null) {
362      return target;
363    }
364
365    target = new AugmentationMethodFinder(invocation, lookup).find();
366    if (target != null) {
367      return target;
368    }
369    return null;
370  }
